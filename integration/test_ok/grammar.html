Grammar

grammar-file:
  ruleset*


ruleset: (used by grammar-file)
  comment newline+ ( rule | newline )+


rule: (used by ruleset)
  identifier : expr-choice








Expression

expr-choice: (used by rule, expr-group)
  expr-sequence ( | expr-sequence )*


expr-sequence: (used by expr-choice)
  expr-negate+


expr-negate: (used by expr-sequence)
  ~? expr-quantified


expr-quantified: (used by expr-negate)
  expr-group quantifier?


expr-group: (used by expr-quantified)
  ( expr-choice ) | expr-primary


expr-primary: (used by expr-group)
  literal-string | regex | identifier








Lexical Grammar

quantifier: (used by expr-quantified)
  ? | + | *


newline: (used by ruleset)
  [ \t\n]* [\n]


comment: (used by ruleset)
  # ~[\n]


identifier: (used by rule, expr-primary)
  [a-zA-Z] [a-zA-Z-]*


literal-string: (used by expr-primary)
  " ~["] "


regex: (used by expr-primary)
  ~? [ ~[\\[] | ] [+?*]?
